  int age=25;
  printf(age); //incorrect
  printf("%d",age);
  return 0;
}

output:
[2021ict37@fedora ~]$ ./secondc
25[2021ict37@fedora ~]$

explanation of command:
printf(age); is incorrect because printf expects a format string, not a variable directly.
printf("%d", age);: Correct way – %d is the format specifier for an integer.


###########################
%d / %i - integers
%f / %F - floats
%lf - char
%s - string

int num1;
int num2;

printf("Enter a num1:");

int sum=num1+num2;
int sub=num1-num2;

int div=num1/num2;

printf("Multiplication of %d * %d: %d/n",num1, num2,mul);
printf("Division of %d / %d: %.2f/n",num1, num2,div);

}
