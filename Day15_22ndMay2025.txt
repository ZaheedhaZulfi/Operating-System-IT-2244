/*print number from 1 to 10
1 to 5 should be print by child precess and
from 6 to 10 should be parent process 
calculate summation of those numbers
*/
#include<stdio.h>
#include<unistd.h>

int main(){
f=fork();
int sum=0;
if(f==0){
for(int i=1; i<=5; i++){
 printf("%d\n", i);	
 sum=sum+i;
}
}
else{
for(int i=6; i<=10; i++){
	printf("%d\n", i);
	 sum=sum+i;
}

}

}
